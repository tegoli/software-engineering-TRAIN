     * UC2 - Login (FR28)
     * @param {Object} req - Express request
     * @param {Object} res - Express response
     */
    static async login(req, res) {
        const { email, password } = req.body;
        const db = readDB();
        const user = db.users.find(u => u.email === email);
        if (!user || user.passwordHash !== hashPassword(password)) {
            // Extension: failed attempts (NFR20)
            if (user) {
                user.failedLoginAttempts = (user.failedLoginAttempts || 0) + 1;
                if (user.failedLoginAttempts >= 5) {
                    user.accountStatus = 'blocked';
                }
                writeDB(db);
            }
            return res.status(401).json({ success: false, message: 'Credenziali errate' });
        }
        if (user.accountStatus !== 'active') {
            return res.status(401).json({ success: false, message: 'Account bloccato' });
        }
        // Reset tentativi falliti
        user.failedLoginAttempts = 0;
        writeDB(db);

        const token = generateToken();
        sessions.set(token, { userId: user.userId, role: user.role });
        res.json({
            success: true,
            token,
            user: { id: user.userId, name: user.name, role: user.role }
        });
    }

    static async logout(req, res) {

    static authenticate(req, res, next) {
        const auth = req.headers.authorization;
        if (!auth || !auth.startsWith('Bearer ')) {
            return res.status(401).json({ error: 'Unauthorized' });
        }
        const token = auth.slice(7);
        const session = sessions.get(token);
        if (!session) {
            return res.status(401).json({ error: 'Invalid token' });
        }
        req.user = session;
        next();
    }

function calculatePrice(fromStation, toStation, travelClass) {
    const R = 6371;
    const dLat = (toStation.latitude - fromStation.latitude) * Math.PI / 180;
    const dLon = (toStation.longitude - fromStation.longitude) * Math.PI / 180;
    const a = Math.sin(dLat/2)**2 + Math.cos(fromStation.latitude * Math.PI/180) * Math.cos(toStation.latitude * Math.PI/180) * Math.sin(dLon/2)**2;
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    const km = R * c;
    let price = km * 0.15;
    if (travelClass === 'business') price *= 1.8;
    return Math.round(price * 100) / 100;
}

export class TrainSearchController {
    /**
     * UC1 - Search Train (FR1, FR3, FR4, FR5, FR11)
     * @param {Object} req - body: { departureStationName, arrivalStationName, date, travelClass }
     * @param {Object} res
     */
    static async search(req, res) {
        const { departureStationName, arrivalStationName, date, travelClass } = req.body;
        const db = readDB();

        const fromStation = db.stations.find(s => s.name.toLowerCase() === departureStationName.toLowerCase());
        const toStation = db.stations.find(s => s.name.toLowerCase() === arrivalStationName.toLowerCase());
        if (!fromStation || !toStation) {
            return res.json({ trains: [] });
        }

        const searchDate = new Date(date);
        const results = [];

        for (const run of db.trainRuns) {
            const route = db.routes.find(r => r.routeId === run.routeId);
            if (!route) continue;
            const fromStop = route.stops.find(stop => stop.stationId === fromStation.stationId);
            const toStop = route.stops.find(stop => stop.stationId === toStation.stationId);
            if (!fromStop || !toStop || fromStop.stopOrder >= toStop.stopOrder) continue;

            const runDate = new Date(run.departureDateTime);
            if (runDate.toDateString() !== searchDate.toDateString()) continue;
            if (run.status === 'cancelled') continue;

            const train = db.trains.find(t => t.trainId === run.trainId);
            if (!train.serviceClasses.includes(travelClass)) continue;

            const depTime = new Date(runDate);
            const [depHour, depMin] = fromStop.scheduledDeparture.split(':');
            depTime.setHours(parseInt(depHour), parseInt(depMin), 0);

            const arrTime = new Date(runDate);
            const [arrHour, arrMin] = toStop.scheduledArrival.split(':');
            arrTime.setHours(parseInt(arrHour), parseInt(arrMin), 0);
            if (arrTime < depTime) arrTime.setDate(arrTime.getDate() + 1);

        const train = db.trains.find(t => t.trainId === run.trainId);

            const station = db.stations.find(s => s.stationId === stop.stationId);

    static async stationBoard(req, res) {

        if (!station) return res.status(404).json({ error: 'Station not found' });

        const today = new Date().toISOString().slice(0,10);
        const results = [];

        for (const run of db.trainRuns) {
            const runDate = new Date(run.departureDateTime).toISOString().slice(0,10);
            if (runDate !== today) continue;

            const route = db.routes.find(r => r.routeId === run.routeId);
            const stop = route.stops.find(s => s.stationId === station.stationId);
            if (!stop) continue;

            const train = db.trains.find(t => t.trainId === run.trainId);
            if (boardType === 'departures' && stop.scheduledDeparture) {
                results.push({
                    trainCode: train.trainCode,
                    destination: route.stops[route.stops.length-1].stationName,
                    scheduledTime: stop.scheduledDeparture,
                    track: stop.trackNumber,
                    status: run.status
                });

                    trainCode: train.trainCode,
                    origin: route.stops[0].stationName,
                    scheduledTime: stop.scheduledArrival,
                    track: stop.trackNumber,
                    status: run.status
                });

/**
 * @class Route
 * @description Percorso di una linea ferroviaria.
 */
export class Route {
    constructor(routeId, routeName, stops) {
        this.routeId = routeId;
        this.routeName = routeName;
        this.stops = stops; // array di oggetti Stop
    }

    /**
     * UC1 - Calcola la durata totale del percorso.
     * @returns {number} Minuti totali.
     */
    calculateDuration() {
        if (!this.stops || this.stops.length < 2) return 0;
        const firstStop = this.stops[0];
        const lastStop = this.stops[this.stops.length - 1];
        // Supponiamo che scheduledDeparture e scheduledArrival siano stringhe HH:MM
        const start = firstStop.scheduledDeparture ? firstStop.scheduledDeparture : firstStop.scheduledArrival;
        const end = lastStop.scheduledArrival ? lastStop.scheduledArrival : lastStop.scheduledDeparture;
        if (!start || !end) return 0;
        const [sh, sm] = start.split(':');
        const [eh, em] = end.split(':');
        let total = (parseInt(eh) * 60 + parseInt(em)) - (parseInt(sh) * 60 + parseInt(sm));
        if (total < 0) total += 24 * 60;
        return total;
    }
}

/**
 * @class Location
 * @description Coordinate geografiche (latitudine, longitudine).
 */
export class Location {
    /**
     * @param {number} lat - Latitudine.
     * @param {number} lon - Longitudine.
     */
    constructor(lat, lon) {
        this.lat = lat;
        this.lon = lon;
    }

    /**
     * Calcola la distanza approssimativa in km da un'altra posizione.
     * @param {Location} other
     * @returns {number}
     */
    distanceTo(other) {
        const R = 6371;
        const dLat = (other.lat - this.lat) * Math.PI / 180;
        const dLon = (other.lon - this.lon) * Math.PI / 180;
        const a = Math.sin(dLat/2)**2 + Math.cos(this.lat * Math.PI/180) * Math.cos(other.lat * Math.PI/180) * Math.sin(dLon/2)**2;
        const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        return R * c;
    }
}
